57  template<class NumericType>
58  inline
59  void read_species_data_ascii( ChemicalMixture<NumericType>& chem_mixture,
60  std::istream& in )
61  {
62  skip_comment_lines(in, '#');
63 
64  std::string name;
65  NumericType mol_wght, h_form, n_tr_dofs;
66  int charge;
67 
68  while (in.good())
69  {
70  in >> name; // Species Name
71  in >> mol_wght; // molecular weight (kg/kmol)
72  in >> h_form; // heat of formation at Ok (J/kg)
73  in >> n_tr_dofs; // number of translational/rotational DOFs
74  in >> charge; // charge number
75 
76  // If we are still good, we have a valid set of thermodynamic
77  // data for this species. Otherwise, we read past end-of-file
78  // in the section above
79  if (in.good())
80  {
81  // If we do not have an enum for this species, that is going to cause
82  // problems down the line.
83  if (!chem_mixture.species_name_map().count(name))
84  {
85  std::cerr << "ERROR: Unexpected species " << name
86  << " encountered while parsing thermodynamic table!"
87  << std::endl;
88  antioch_error();
89  }
90 
91  /* Only add a ChemicalSpecies if the user supplied it to
92  the ChemicalMixture. */
93  Species species = chem_mixture.species_name_map().find(name)->second;
94 
95  // using default comparison:
96  std::vector<Species>::const_iterator it = std::search_n( chem_mixture.species_list().begin(),
97  chem_mixture.species_list().end(),
98  1, species);
99  if( it != chem_mixture.species_list().end() )
100  {
101  unsigned int index = static_cast<unsigned int>(it - chem_mixture.species_list().begin());
102  chem_mixture.add_species( index, name, mol_wght, h_form, n_tr_dofs, charge );
103  }
104 
105  }
106  }
107 
108  return;
109  }

207  std::istream &in)
208  {

214  unsigned int n_degeneracies;
215 
216  while (in.good())
217  {
218  in >> name; // Species Name
219  in >> theta_v; // characteristic vibrational temperature (K)
220  in >> n_degeneracies; // degeneracy of the mode
221 
222  // If we are still good, we have a valid set of thermodynamic
223  // data for this species. Otherwise, we read past end-of-file
224  // in the section above
225  if (in.good())
226  {
227  // If we do not have an enum for this species, that is going to cause
228  // problems down the line.
229  if (!chem_mixture.species_name_map().count(name))
230  {
231  std::cerr << "ERROR: Unexpected species " << name
232  << " encountered while parsing thermodynamic table!"
233  << std::endl;
234  antioch_error();
235  }
236 
237  // If this species is not part of the existing mixture, we
238  // don't care about it, so just continue silently
239  if( !chem_mixture.active_species_name_map().count(name) )
240  {
241  continue;
242  }
243 
244  // ... otherwise we add the data
245  const unsigned int s =
246  chem_mixture.active_species_name_map().find(name)->second;
247 
248  antioch_assert_equal_to((chem_mixture.chemical_species()[s])->species(), name);
249 
250  chem_mixture.add_species_vibrational_data(s, theta_v, n_degeneracies);
251  }
252 
253  }
