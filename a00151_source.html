49  template<typename CoeffType>
50  inline
51  Reaction<CoeffType>* build_reaction( const unsigned int n_species,
52  const std::string& equation,
53  const ReactionType::ReactionType& type ,
54  const KineticsModel::KineticsModel& kin )
55  {
56  Reaction<CoeffType>* reaction = NULL;
57 
58  switch(type)
59  {
60  case(ReactionType::ELEMENTARY):
61  {
62  reaction = new ElementaryReaction<CoeffType>(n_species,equation,kin);
63  }
64  break;
65 
66  case(ReactionType::DUPLICATE):
67  {
68  reaction = new DuplicateReaction<CoeffType>(n_species,equation,kin);
69  }
70  break;
71 
72  case(ReactionType::THREE_BODY):
73  {
74  reaction = new ThreeBodyReaction<CoeffType>(n_species,equation,kin);
75  }
76  break;
77 
78  case(ReactionType::LINDEMANN_FALLOFF):
79  {
80  reaction = new FalloffReaction<CoeffType,LindemannFalloff<CoeffType> >(n_species,equation,type,kin);
81  }
82  case(ReactionType::TROE_FALLOFF):
83  {
84  reaction = new FalloffReaction<CoeffType,TroeFalloff<CoeffType> >(n_species,equation,type,kin);
85  }
86  break;
87 
88  default:
89  {
90  antioch_error();
91  }
92 
93  } // switch(type)
94 
95  // Dummy
96  return reaction;
97  }
