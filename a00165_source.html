50  template<class NumericType>
51  inline
52  void read_sutherland_data_ascii( MixtureViscosity<SutherlandViscosity<NumericType>,NumericType >& mu,
53  std::istream &in)
54  {
55  // skip the header
56  skip_comment_lines(in, '#');
57 
58  std::string name;
59  NumericType a, b;
60 
61  while (in.good())
62  {
63  in >> name; // Species Name
64  in >> a; //
65  in >> b; //
66 
67  // If we are still good, we have a valid set of transport
68  // data for this species. Otherwise, we read past end-of-file
69  // in the section above
70  if (in.good())
71  {
72  const ChemicalMixture<NumericType>& chem_mixture = mu.chemical_mixture();
73 
74  // Check if this is a species we want.
75  if( chem_mixture.active_species_name_map().find(name) !=
76  chem_mixture.active_species_name_map().end() )
77  {
78  // Pack up coefficients
79  std::vector<NumericType> coeffs(2);
80  coeffs[0] = a;
81  coeffs[1] = b;
82  mu.add(name, coeffs);
83  }
84  }
85  }
86 
87  return;
88  }
