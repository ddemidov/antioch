308  os << " -----------------------------\n"

317 
318  for (unsigned int l=0; l<this->theta_v().size(); l++)
319  os << " theta_v_" << l << " = " << this->theta_v()[l]
320  << ", ndg = " << this->ndg_v()[l] << "\n";
321 
322  for (unsigned int l=0; l<this->theta_e().size(); l++)
323  os << " theta_e_" << l << " = " << this->theta_e()[l]
324  << ", ndg = " << this->ndg_e()[l] << "\n";
325 
326  os << '\n';
327 
328  return;
