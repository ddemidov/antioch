57  template<class NumericType>
58  inline
59  void read_cea_mixture_data_ascii( CEAThermoMixture<NumericType>& thermo, std::istream &in )
60  {
61  skip_comment_lines(in, '#');
62 
63  std::string name;
64  unsigned int n_int;
65  std::vector<NumericType> coeffs;
66  NumericType h_form, val;
67 
68  const ChemicalMixture<NumericType>& chem_mixture = thermo.chemical_mixture();
69 
70  while (in.good())
71  {
72  in >> name; // Species Name
73  in >> n_int; // Number of T intervals: [200-1000], [1000-6000], ([6000-20000])
74  in >> h_form; // Formation Enthalpy @ 298.15 K
75 
76  coeffs.clear();
77  for (unsigned int interval=0; interval<n_int; interval++)
78  {
79  for (unsigned int n=0; n<10; n++)
80  {
81  in >> val, coeffs.push_back(val);
82  }
83  }
84 
85  // If we are still good, we have a valid set of thermodynamic
86  // data for this species. Otherwise, we read past end-of-file
87  // in the section above
88  if (in.good())
89  {
90  // Check if this is a species we want.
91  if( chem_mixture.active_species_name_map().find(name) !=
92  chem_mixture.active_species_name_map().end() )
93  {
94  thermo.add_curve_fit(name, coeffs);
95  }
96  }
97  } // end while
98 
99  // Make sure we actually populated everything
100  if( !thermo.check() )
101  {
102  std::cerr << "Error: CEA table not fully populated" << std::endl;
103  antioch_error();
104  }
105 
106  return;
107  }
