248  const VectorStateType& mass_fractions,

250  {
251  antioch_assert_equal_to( mass_fractions.size(), this->n_species() );

257  for( unsigned int s = 0; s < this->n_species(); s++ )

265  template<typename CoeffType>
266  inline
267  ChemicalMixture<CoeffType>::ChemicalMixture( const std::vector<std::string>& species_list )
268  : _chemical_species( species_list.size(), NULL )
269  {
270  // Build up name map for all possible species
271  this->init_species_name_map();
272 
273  // Build up inverse name map
274  this->build_inverse_name_map();
275 
276  // Populate species list for requested species
277  _species_list.reserve( species_list.size() );
278  for( unsigned int s = 0; s < species_list.size(); s++ )
279  {
280  if( _species_name_map.find( species_list[s] ) == _species_name_map.end() )
281  {
282  std::cerr << "Error in ChemicalMixture: Unknown species " << species_list[s] << std::endl;
283  antioch_error();
284  }
285 
286  _species_list.push_back( _species_name_map.find( species_list[s] )->second );
287  _species_list_map.insert( std::make_pair( _species_name_map.find( species_list[s] )->second, s ) );
288  _active_species_name_map.insert( std::make_pair( species_list[s], s ) );
289  }
290 
291  // Now read in chemical properties for the requested species and stash
292  read_species_data_ascii_default(*this);
293 
294  //... and any vibrational data
295  read_species_vibrational_data_ascii_default(*this);
296 
297  //... and any electronic data
298  read_species_electronic_data_ascii_default(*this);
299  }

317  template<typename CoeffType>
318  template<typename VectorStateType>
319  inline
320  typename enable_if_c<
321  has_size<VectorStateType>::value,
322  typename Antioch::value_type<VectorStateType>::type
323  >::type
324  ChemicalMixture<CoeffType>::R( const VectorStateType& mass_fractions ) const
325  {
326  antioch_assert_equal_to( mass_fractions.size(), _chemical_species.size() );
327  antioch_assert_greater( mass_fractions.size(), 0);
328 
329  typename Antioch::value_type<VectorStateType>::type
330  R = mass_fractions[0]*this->R(0);
331  for( unsigned int s = 1; s < mass_fractions.size(); s++ )
332  {
333  R += mass_fractions[s]*this->R(s);
334  }
335 
336  return R;
337  }

340  template<typename CoeffType>
341  template<typename VectorStateType>
342  inline
343  typename enable_if_c<
344  has_size<VectorStateType>::value,
345  typename Antioch::value_type<VectorStateType>::type
346  >::type
347  ChemicalMixture<CoeffType>::M( const VectorStateType& mass_fractions ) const
348  {
349  antioch_assert_equal_to( mass_fractions.size(), _chemical_species.size() );
350 
351  typename Antioch::value_type<VectorStateType>::type
352  M = mass_fractions[0]/this->M(0);
353  for( unsigned int s = 1; s < mass_fractions.size(); s++ )
354  {
355  M += mass_fractions[s]/(this->M(s));
356  }
357 
358  return CoeffType(1)/M;
359  }
