116  StateType ThreeBodyReaction<CoeffType>::compute_forward_rate_coefficient( const VectorStateType& molar_densities,
117  const StateType& T ) const
118  {
119  //k(T,[M]) = (sum eff_i * C_i) * ...
120  StateType kfwd = (this->efficiency(0) * molar_densities[0] );
121 
122  for (unsigned int s=1; s<this->n_species(); s++)
123  {
124  kfwd += ( this->efficiency(s) * molar_densities[s] );
125  }
126 
127  //... alpha(T)
128  return (kfwd * (*this->_forward_rate[0])(T));
129  }

134  inline
135  void ThreeBodyReaction<CoeffType>::compute_forward_rate_coefficient_and_derivatives( const VectorStateType &molar_densities,
136  const StateType& T,
137  StateType& kfwd,
138  StateType& dkfwd_dT,
139  VectorStateType& dkfwd_dX) const
140  {

145  this->_forward_rate[0]->compute_rate_and_derivative(T,kfwd,dkfwd_dT);
146

149 
150  for (unsigned int s=1; s<this->n_species(); s++)

153  dkfwd_dX[s] = kfwd;

162  }
