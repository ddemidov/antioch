536  template<typename CoeffType>
537  inline
538  void Reaction<CoeffType>::initialize()
539  {
540  // Stoichiometric coefficients, by species id
541  {
542  _species_reactant_stoichiometry.resize(this->n_species());
543  _species_product_stoichiometry.resize(this->n_species());
544  _species_delta_stoichiometry.resize(this->n_species());
545 
546  std::fill( _species_reactant_stoichiometry.begin(),
547  _species_reactant_stoichiometry.end(),
548  0 );
549 
550  std::fill( _species_product_stoichiometry.begin(),
551  _species_product_stoichiometry.end(),
552  0);
553  }
554 
555  for (unsigned int r=0; r< this->n_reactants(); r++)
556  {
557  _species_reactant_stoichiometry[this->reactant_id(r)] =
558  this->reactant_stoichiometric_coefficient(r);
559  }
560 
561  for (unsigned int p=0; p < this->n_products(); p++)
562  {
563  _species_product_stoichiometry[this->product_id(p)] =
564  this->product_stoichiometric_coefficient(p);
565  }
566 
567  // find the delta stoichiometric coefficient for each species,
568  // and the sum of the deltas
569  _gamma = 0;
570  for (unsigned int s=0; s<this->n_species(); s++)
571  {
572  _species_delta_stoichiometry[s] =
573  ( _species_product_stoichiometry[s] -
574  _species_reactant_stoichiometry[s] );
575 
576  _gamma += _species_delta_stoichiometry[s];
577  }
578 
579  // set initialization flag
580  _initialized = true;
581  }

584  template<typename CoeffType>
585  template<typename StateType, typename VectorStateType>
586  inline
587  StateType Reaction<CoeffType>::equilibrium_constant( const StateType& P0_RT,
588  const VectorStateType& h_RT_minus_s_R ) const
589  {
590  using std::exp;
591 
592  antioch_assert( this->initialized() );
594  // antioch_assert_greater( P0_RT, 0.0 );
595  antioch_assert_greater( h_RT_minus_s_R.size(), 0 );
596  antioch_assert_equal_to( h_RT_minus_s_R.size(), this->n_species() );
597  antioch_assert_equal_to( _species_delta_stoichiometry.size(), this->n_species() );
598 
599  StateType exppower = -( static_cast<CoeffType>(_species_delta_stoichiometry[0])*
600  h_RT_minus_s_R[0] );
601 
602  for (unsigned int s=1; s < this->n_species(); s++)
603  {
604  exppower += -( static_cast<CoeffType>(_species_delta_stoichiometry[s])*
605  h_RT_minus_s_R[s] );
606  }
607 
608  return ant_pow( P0_RT, static_cast<CoeffType>(this->gamma()) )*exp(exppower);
609  }

612  template<typename CoeffType>
613  template<typename StateType, typename VectorStateType>
614  inline
615  void Reaction<CoeffType>::equilibrium_constant_and_derivative( const StateType& T,
616  const StateType& P0_RT,
617  const VectorStateType& h_RT_minus_s_R,
618  const VectorStateType& ddT_h_RT_minus_s_R,
619  StateType& keq,
620  StateType& dkeq_dT) const
621  {
622  antioch_assert(this->initialized());
623 
625  // antioch_assert_greater( P0_RT, 0.0 );
626  // antioch_assert_greater( T, 0.0 );
627  antioch_assert_greater( h_RT_minus_s_R.size(), 0 );
628  antioch_assert_equal_to( h_RT_minus_s_R.size(), this->n_species() );
629  antioch_assert_equal_to( ddT_h_RT_minus_s_R.size(), this->n_species() );
630  antioch_assert_equal_to( _species_delta_stoichiometry.size(), this->n_species() );
631 
632  // get the equilibrium constant
633  keq = this->equilibrium_constant( P0_RT, h_RT_minus_s_R );
634 
635  StateType ddT_exppower = -( static_cast<CoeffType>(_species_delta_stoichiometry[0])*
636  ddT_h_RT_minus_s_R[0] );
637 
638  for (unsigned int s=1; s<this->n_species(); s++)
639  ddT_exppower += -( static_cast<CoeffType>(_species_delta_stoichiometry[s])*
640  ddT_h_RT_minus_s_R[s] );
641 
642  // compute its derivative
643  dkeq_dT = keq*(-static_cast<CoeffType>(this->gamma())/T + ddT_exppower);
644 
645  return;
646  }

754  (static_cast<const ThreeBodyReaction<CoeffType>*>(this))->compute_forward_rate_coefficient_and_derivatives(molar_densities,T,kfwd,dkfwd_dT,dkfwd_dX);
755  }
756  break;
757 
758  case(ReactionType::LINDEMANN_FALLOFF):
759  {
760  (static_cast<const FalloffReaction<CoeffType,LindemannFalloff<CoeffType> >*>(this))->compute_forward_rate_coefficient_and_derivatives(molar_densities,T,kfwd,dkfwd_dT,dkfwd_dX);
761  }
762  break;

764  case(ReactionType::TROE_FALLOFF):
765  {
766  (static_cast<const FalloffReaction<CoeffType,TroeFalloff<CoeffType> >*>(this))->compute_forward_rate_coefficient_and_derivatives(molar_densities,T,kfwd,dkfwd_dT,dkfwd_dX);
767  }
768  break;

775  } // switch(type)

780  //kfwd *prod_r [R]^nu_r - kbkwd * prod_p [P]^nu_p ( = - 1/nu_r d[R]/dt)
781  template<typename CoeffType>
782  template <typename StateType, typename VectorStateType>
783  inline
784  StateType Reaction<CoeffType>::compute_rate_of_progress( const VectorStateType& molar_densities,
785  const StateType& T,
786  const StateType& P0_RT,
787  const VectorStateType& h_RT_minus_s_R) const
788  {
789  StateType kfwd_times_reactants = this->compute_forward_rate_coefficient(molar_densities,T);
790  StateType Keq = this->equilibrium_constant( P0_RT, h_RT_minus_s_R );
791  StateType kbkwd_times_products = kfwd_times_reactants/Keq;
792  // Rfwd
793  for (unsigned int ro=0; ro < this->n_reactants(); ro++)
794  {
795  kfwd_times_reactants *=
796  ant_pow( molar_densities[this->reactant_id(ro)],
797  static_cast<int>(this->reactant_stoichiometric_coefficient(ro)) );
798  }
799 
800  // Rbkwd
801  for (unsigned int po=0; po< this->n_products(); po++)
802  {
803  kbkwd_times_products *=
804  ant_pow( molar_densities[this->product_id(po)],
805  static_cast<int>(this->product_stoichiometric_coefficient(po)) );
806 
807  }
808 
809  return kfwd_times_reactants - kbkwd_times_products;
810 
811  }

813  template<typename CoeffType>
814  template <typename StateType, typename VectorStateType>
815  inline
816  void Reaction<CoeffType>::compute_rate_of_progress_and_derivatives( const VectorStateType &molar_densities,
817  const ChemicalMixture<CoeffType>& chem_mixture,
818  const StateType& T,
819  const StateType &P0_RT,
820  const VectorStateType &h_RT_minus_s_R,
821  const VectorStateType &dh_RT_minus_s_R_dT,
822  StateType& net_reaction_rate,
823  StateType& dnet_rate_dT,
824  VectorStateType& dnet_rate_dX_s ) const
825  {
826  antioch_assert_equal_to (molar_densities.size(), this->n_species());
827 
828 
829  StateType kfwd = Antioch::zero_clone(T);
830  StateType dkfwd_dT = Antioch::zero_clone(T);
831  VectorStateType dkfwd_dX_s = Antioch::zero_clone(molar_densities);
832  VectorStateType dkbkwd_dX_s = Antioch::zero_clone(molar_densities);
833 
834  compute_forward_rate_coefficient_and_derivatives(molar_densities, T, kfwd, dkfwd_dT ,dkfwd_dX_s);
835 
836  StateType keq = Antioch::zero_clone(T);
837  StateType dkeq_dT = Antioch::zero_clone(T);
838 
839  equilibrium_constant_and_derivative( T, P0_RT, h_RT_minus_s_R,
840  dh_RT_minus_s_R_dT,
841  keq, dkeq_dT );
842 
843  const StateType kbkwd = kfwd/keq;
844  const StateType dkbkwd_dT = (dkfwd_dT - kbkwd*dkeq_dT)/keq;
845  for(unsigned int s = 0; s < this->n_species(); s++)
846  {
847  dkbkwd_dX_s[s] = dkfwd_dX_s[s]/keq;
848  }
849 
850  // If users want to use valarrays, then the output reference sizes
851  // had better already match the input value sizes...
852 
853  StateType Rfwd = Antioch::zero_clone(kfwd);
854  StateType dRfwd_dT = Antioch::zero_clone(kbkwd);
855  StateType Rbkwd = Antioch::zero_clone(dkfwd_dT);
856  StateType dRbkwd_dT = Antioch::zero_clone(dkbkwd_dT);
857 
858  // We need to construct using an input StateType argument if we
859  // want StateType==valarray to have the right sizes
860  // valarray compatibility makes this a bit redundant, but not much
861  // worse than the previous version
863  VectorStateType dRfwd_dX_s(this->n_species(), kfwd);
864  VectorStateType dRbkwd_dX_s(this->n_species(), kbkwd);
865 
866  Antioch::set_zero(dRfwd_dX_s);
867  Antioch::set_zero(dRbkwd_dX_s);
868 
869  // pre-fill the participating species partials with the rates
870  for (unsigned int r=0; r< this->n_reactants(); r++)
871  {
872  dRfwd_dX_s[this->reactant_id(r)] = kfwd;
873  }
874 
875  for (unsigned int p=0; p < this->n_products(); p++)
876  {
877  dRbkwd_dX_s[this->product_id(p)] = kbkwd;
878  }
879 
880  //init
881  Rfwd = kfwd;
882  dRfwd_dT = dkfwd_dT;
883  Rbkwd = kbkwd;
884  dRbkwd_dT = dkbkwd_dT;
885 
886  // Rfwd & derivatives
887  for (unsigned int ro=0; ro < this->n_reactants(); ro++)
888  {
889  const StateType val =
890  ant_pow( molar_densities[this->reactant_id(ro)],
891  static_cast<int>(this->reactant_stoichiometric_coefficient(ro)) );
892 
893  const StateType dval =
894  ( static_cast<CoeffType>(this->reactant_stoichiometric_coefficient(ro))*
895  ant_pow( molar_densities[this->reactant_id(ro)],
896  static_cast<int>(this->reactant_stoichiometric_coefficient(ro))-1 )
897  );
898 
899  Rfwd *= val;
900  dRfwd_dT *= val;
901 
902  for (unsigned int ri=0; ri<this->n_reactants(); ri++)
903  {
904  dRfwd_dX_s[this->reactant_id(ri)] *= (ri == ro) ? dval : val;
905  }
906  }
907 
908  const StateType facfwd = Rfwd/kfwd;
909  for (unsigned int s = 0; s < this->n_species(); s++)
910  {
911  dRfwd_dX_s[s] += facfwd * dkfwd_dX_s[s];
912  }
913 
914 
915  // Rbkwd & derivatives
916  for (unsigned int po=0; po< this->n_products(); po++)
917  {
918  const StateType val =
919  ant_pow( molar_densities[this->product_id(po)],
920  static_cast<int>(this->product_stoichiometric_coefficient(po)) );
921 
922  const StateType dval =
923  ( static_cast<CoeffType>(this->product_stoichiometric_coefficient(po))*
924  ant_pow( molar_densities[this->product_id(po)],
925  static_cast<int>(this->product_stoichiometric_coefficient(po))-1 )
926  );
927 
928  Rbkwd *= val;
929  dRbkwd_dT *= val;
930 
931  for (unsigned int pi=0; pi<this->n_products(); pi++)
932  {
933  dRbkwd_dX_s[this->product_id(pi)] *= (pi == po) ? dval : val;
934  }
935 
936  }
937 
938  const StateType facbkwd = Rbkwd/kbkwd;
939 
940  for (unsigned int s = 0; s < this->n_species(); s++)
941  {
942  dRbkwd_dX_s[s] += facbkwd * dkbkwd_dX_s[s];
943  }
944 
945  net_reaction_rate = Rfwd - Rbkwd;
946 
947  dnet_rate_dT = dRfwd_dT - dRbkwd_dT;
948 
949  for (unsigned int s = 0; s < this->n_species(); s++)
950  {
951  dnet_rate_dX_s[s] = dRfwd_dX_s[s] - dRbkwd_dX_s[s];
952  }
953 
954  return;
955  }
