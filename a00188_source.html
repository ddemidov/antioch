71  template<class CoeffType>
72  WilkeMixture<CoeffType>::WilkeMixture( const ChemicalMixture<CoeffType>& chem_mixture )
73  : _chem_mixture(chem_mixture),
74  _Mr_Ms_to_the_one_fourth(chem_mixture.n_species()),
75  _denom(chem_mixture.n_species())
76  {
77  using std::pow;
78 
79  for( unsigned int r = 0; r < chem_mixture.n_species(); r++ )
80  {
81  _Mr_Ms_to_the_one_fourth[r].resize(chem_mixture.n_species());
82  _denom[r].resize(chem_mixture.n_species());
83 
84  for( unsigned int s = 0; s < chem_mixture.n_species(); s++ )
85  {
86  const CoeffType Mr = chem_mixture.M(r);
87  const CoeffType Ms = chem_mixture.M(s);
88 
89  _Mr_Ms_to_the_one_fourth[r][s] = pow( Mr/Ms, CoeffType(0.25) );
90  _denom[r][s] = std::sqrt(8.0*(1.0+Ms/Mr));
91  }
92  }
93 
94  return;
95  }
