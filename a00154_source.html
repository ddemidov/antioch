196  template<typename CoeffType>
197  template<typename StateType, typename VectorStateType>
198  inline
199  StateType TroeFalloff<CoeffType>::operator()(const StateType& T,
200  const VectorStateType &molar_densities,
201  const StateType &k0,
202  const StateType &kinf) const
203  {
204 
205  //compute log(Fcent) once
206  StateType logFcent = ant_log(this->Fcent(T));
207  //[M]
208  StateType M = Antioch::zero_clone(T);
209  for(unsigned int i = 0; i < molar_densities.size(); i++)
210  {
211  M += molar_densities[i];
212  }
213 
214  // Pr = [M] * k0/kinf
215  ANTIOCH_AUTO(StateType) Pr = M * k0/kinf;
216  // c = -0.4 - 0.67 * log10(Fcent)
217  // Note log10(x) = (1.0/log(10))*log(x)
218  StateType c = - 0.4 - _c_coeff*logFcent;
219 
220  // n = 0.75 - 1.27 * log10(Fcent)
221  // Note log10(x) = (1.0/log(10))*log(x)
222  ANTIOCH_AUTO(StateType) n = 0.75 - _n_coeff*logFcent;
223  ANTIOCH_AUTO(StateType) d = Antioch::constant_clone(T, 0.14);
224 
225  StateType log10Pr = Constants::log10_to_log<CoeffType>() * ant_log(Pr);
226 
227  //log10F = log10(Fcent) / [1+((log10(Pr) + c)/(n - d*(log10(Pr) + c) ))^2]
228  //logF = log(Fcent) / [1+((log10(Pr) + c)/(n - d*(log10(Pr) + c) ))^2]
229  ANTIOCH_AUTO(StateType) logF =
230  logFcent/(1. + ant_pow(((log10Pr + c)/(n - d*(log10Pr + c) )),2) );
231 
232  return ant_exp(logF);
233  }

238  inline
239  StateType TroeFalloff<CoeffType>::Fcent(const StateType &T) const
240  {
241 
242  // Fcent = (1.-alpha)*exp(-T/T***) + alpha * exp(-T/T*) + exp(-T**/T)
243  StateType Fc = (1. - _alpha) * ant_exp(-T/_T3) + _alpha * ant_exp(-T/_T1);
244 
245  if(_T2 != 1e50)Fc += ant_exp(-_T2/T);
246 
247  return Fc;
248  }

252  inline
253  void TroeFalloff<CoeffType>::Fcent_and_derivatives(const StateType &T, StateType &Fc, StateType &dFc_dT) const
254  {
255 
256  // Fcent = (1.-alpha)*exp(-T/T***) + alpha * exp(-T/T*) + exp(-T**/T)
257  Fc = (1. - _alpha) * ant_exp(-T/_T3) + _alpha * ant_exp(-T/_T1);
258  dFc_dT = (_alpha - 1.)/_T3 * ant_exp(-T/_T3) - _alpha/_T1 * ant_exp(-T/_T1);
259 
260  if(_T2 != 1e50)
261  {
262  Fc += ant_exp(-_T2/T);
263  dFc_dT += _T2/(T*T) * ant_exp(-_T2/T);
264  }
265 
266  return;
267  }

270  template <typename StateType, typename VectorStateType>
271  inline
272  void TroeFalloff<CoeffType>::F_and_derivatives(const StateType& T,
273  const VectorStateType &molar_densities,
274  const StateType &k0,
275  const StateType &dk0_dT,
276  const StateType &kinf,
277  const StateType &dkinf_dT,
278  StateType &F,
279  StateType &dF_dT,
280  VectorStateType &dF_dX) const
281  {
282 
283  antioch_assert_equal_to(dF_dX.size(),this->n_spec);
284 
285  //declarations
286  //M
287  StateType M = Antioch::zero_clone(T);
288  for(unsigned int i = 0; i < molar_densities.size(); i++)
289  {
290  M += molar_densities[i];
291  }
292  // Pr and derivatives
293  StateType Pr = M * k0/kinf;
294  StateType dPr_dT = Pr * (dk0_dT/k0 - dkinf_dT/kinf);
295  StateType log10Pr = Constants::log10_to_log<CoeffType>() * ant_log(Pr);
296  StateType dlog10Pr_dT = Constants::log10_to_log<CoeffType>()*dPr_dT/Pr;
297  VectorStateType dlog10Pr_dX = Antioch::zero_clone(molar_densities);
298  for(unsigned int ip = 0; ip < dlog10Pr_dX.size(); ip++)
299  {//dlog10Pr_dX = 1/(ln(10)*Pr) * dPr_dX
300  dlog10Pr_dX[ip] = Constants::log10_to_log<CoeffType>()/M; //dPr_dX = k0/kinf, Pr = M k0/kinf => dlog10Pr_dX = 1/(ln(10)*M)
301  }
302  // Fcent and derivatives
303  StateType Fcent = Antioch::zero_clone(T);
304  StateType dFcent_dT = Antioch::zero_clone(T);
305  this->Fcent_and_derivatives(T,Fcent,dFcent_dT);
306  StateType dlog10Fcent_dT = Constants::log10_to_log<CoeffType>()*dFcent_dT/Fcent;
307  // Compute log(Fcent) once
308  StateType logFcent = ant_log(Fcent);
309  // n and c and derivatives
310  StateType d = Antioch::constant_clone(T, 0.14);
311  StateType c = - 0.4 - _c_coeff * logFcent;
312  StateType n = 0.75 - _n_coeff * logFcent;
313  StateType dc_dT = - _c_coeff * dFcent_dT/Fcent;
314  ANTIOCH_AUTO(StateType) dn_dT = - _n_coeff * dFcent_dT/Fcent;
315 
316  //log10F
317  StateType logF = logFcent/(1. + ant_pow(((log10Pr + c)/(n - d*(log10Pr + c) )),2));
318  StateType dlogF_dT = logF * (dlog10Fcent_dT / Fcent
319  - 2. * ant_pow((log10Pr + c)/(n - d * (log10Pr + c)),2)
320  * ((dlog10Pr_dT + dc_dT)/(log10Pr + c) -
321  (dn_dT - d * (dlog10Pr_dT + dc_dT))/(n - d * (log10Pr + c))
322  )
323  / (1. + ant_pow((log10Pr + c)/(n - d * (log10Pr + c)),2))
324  );
325  VectorStateType dlogF_dX = Antioch::zero_clone(molar_densities);
326  for(unsigned int ip = 0; ip < dlog10Pr_dX.size(); ip++)
327  {//dlogF_dX = - logF^2/log(Fcent) * dlog10Pr_dX * (1 - 1/(n - d * (log10Pr + c))) * (log10Pr + c)
328  dlogF_dX[ip] = - ant_pow(logF,2)/logFcent * dlog10Pr_dX[ip] *(1. - 1./(n - d * (log10Pr + c))) * (log10Pr + c);
329  }
330 
331  F = ant_exp(logF);
332  dF_dT = F * dlogF_dT;
333  for(unsigned int ip = 0; ip < dlog10Pr_dX.size(); ip++)
334  {//dF_dX = F * dlogF_dX
335  dF_dX[ip] = F * dlogF_dX[ip];
336  }
337 
338  return;
339  }
