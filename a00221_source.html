42 template <typename Scalar>
43 int test_cp( const std::string& species_name, unsigned int species, Scalar cp_exact, Scalar T,
44  const Antioch::CEAEvaluator<Scalar>& thermo )
45 {
46  using std::abs;
47 
48  int return_flag = 0;
49 
50  const Scalar tol = std::numeric_limits<Scalar>::epsilon() * 5;
51 
52  typedef typename Antioch::template TempCache<Scalar> Cache;
53 
54  const Scalar cp = thermo.cp(Cache(T), species);
55 
56  if( abs( (cp_exact - cp)/cp_exact ) > tol )
57  {
58  std::cerr << "Error: Mismatch in species specific heat."
59  << "\nspecies = " << species_name
60  << "\ncp = " << cp
61  << "\ncp_exact = " << cp_exact
62  << "\ndifference = " << (cp_exact - cp)
63  << "\ntolerance = " << tol
64  << "\nT = " << T << std::endl;
65  return_flag = 1;
66  }
67 
68  return return_flag;
69 }

102 template <typename Scalar>
103 Scalar cp( Scalar T, Scalar a0, Scalar a1, Scalar a2,
104  Scalar a3, Scalar a4, Scalar a5, Scalar a6 )
105 {
106  if( T < 200.1)
107  T = 200.1;
108 
109  return a0/(T*T) + a1/T + a2 + a3*T + a4*(T*T) + a5*(T*T*T) + a6*(T*T*T*T);
110 }
111 
112 template <typename Scalar>
113 Scalar h( Scalar T, Scalar a0, Scalar a1, Scalar a2,
114  Scalar a3, Scalar a4, Scalar a5, Scalar a6,
115  Scalar a8 )
116 {
117  return -a0/(T*T) + a1*std::log(T)/T + a2 + a3*T/2.0 + a4*(T*T)/3.0 + a5*(T*T*T)/4.0 + a6*(T*T*T*T)/5.0 + a8/T;
118 }
