69 template <typename PairScalars>
70 int vectester(const PairScalars& example, const std::string& testname)
71 {
72  using std::abs;
73  using std::exp;
74 
75  typedef typename Antioch::value_type<PairScalars>::type Scalar;
76 
77  const Scalar Cf = 1.4;
78  const Scalar Ea = 5.0;
79 
80  Antioch::ArrheniusRate<Scalar> arrhenius_rate(Cf,Ea,1.);
81 
82  // Construct from example to avoid resizing issues
83  PairScalars T = example;
84  for (unsigned int tuple=0; tuple != ANTIOCH_N_TUPLES; ++tuple)
85  {
86  T[2*tuple] = 1500.1;
87  T[2*tuple+1] = 1600.1;
88  }
89 
90  const Scalar rate_exact0 = Cf*exp(-Ea/1500.1);
91  const Scalar rate_exact1 = Cf*exp(-Ea/1600.1);
92  const Scalar derive_exact0 = Ea/(Scalar(1500.1)*Scalar(1500.1)) * Cf * exp(-Ea/Scalar(1500.1));
93  const Scalar derive_exact1 = Ea/(Scalar(1600.1)*Scalar(1600.1)) * Cf * exp(-Ea/Scalar(1600.1));
94 
95  int return_flag = 0;
96 
97 #ifdef ANTIOCH_HAVE_GRVY
98  gt.BeginTimer(testname);
99 #endif
100 
101  const PairScalars rate = arrhenius_rate(T);
102  const PairScalars deriveRate = arrhenius_rate.derivative(T);
103 
104 #ifdef ANTIOCH_HAVE_GRVY
105  gt.EndTimer(testname);
106 #endif
107 
108  const Scalar tol = std::numeric_limits<Scalar>::epsilon()*10;
109 
110  for (unsigned int tuple=0; tuple != ANTIOCH_N_TUPLES; ++tuple)
111  {
112  if( abs( (rate[2*tuple] - rate_exact0)/rate_exact0 ) > tol )
113  {
114  std::cout << "Error: Mismatch in rate values." << std::endl
115  << "rate(T0) = " << rate[2*tuple] << std::endl
116  << "rate_exact = " << rate_exact0 << std::endl
117  << "difference = " << rate[2*tuple] - rate_exact0 << std::endl;
118 
119  return_flag = 1;
120  break;
121  }
122 
123  if( abs( (rate[2*tuple+1] - rate_exact1)/rate_exact1 ) > tol )
124  {
125  std::cout << "Error: Mismatch in rate values." << std::endl
126  << "rate(T1) = " << rate[2*tuple+1] << std::endl
127  << "rate_exact = " << rate_exact1 << std::endl
128  << "difference = " << rate[2*tuple+1] - rate_exact1 << std::endl;
129 
130  return_flag = 1;
131  break;
132  }
133  }
134  if( abs( (deriveRate[0] - derive_exact0)/derive_exact0 ) > tol )
135  {
136  std::cout << std::scientific << std::setprecision(16)
137  << "Error: Mismatch in rate derivative values." << std::endl
138  << "drate_dT(T0) = " << deriveRate[0] << std::endl
139  << "derive_exact = " << derive_exact0 << std::endl;
140 
141  return_flag = 1;
142  }
143  if( abs( (deriveRate[1] - derive_exact1)/derive_exact1 ) > tol )
144  {
145  std::cout << std::scientific << std::setprecision(16)
146  << "Error: Mismatch in rate derivative values." << std::endl
147  << "drate_dT(T1) = " << deriveRate[1] << std::endl
148  << "derive_exact = " << derive_exact1 << std::endl;
149 
150  return_flag = 1;
151  }
152 
153  std::cout << "Arrhenius rate: " << arrhenius_rate << std::endl;
154 
155  return return_flag;
156 }
