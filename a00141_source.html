162  inline
163  StateType FalloffReaction<CoeffType,FalloffType>::compute_forward_rate_coefficient( const VectorStateType& molar_densities,
164  const StateType& T ) const
165  {
166 //falloff is k(T,[M]) = k0*[M]/(1 + [M]*k0/kinf) * F = k0 * ([M]^-1 + k0 * kinf^-1)^-1 * F
167  StateType M = Antioch::zero_clone(T);
168  for(unsigned int i = 0; i < molar_densities.size(); i++)
169  {
170  M += molar_densities[i];
171  }
172 
173  return (*this->_forward_rate[0])(T) / (ant_pow(M,-1) + (*this->_forward_rate[0])(T) /(*this->_forward_rate[1])(T)) *
174  _F(T,molar_densities,(*this->_forward_rate[0])(T),(*this->_forward_rate[1])(T));
175  }

179  inline
180  void FalloffReaction<CoeffType,FalloffType>::compute_forward_rate_coefficient_and_derivatives( const VectorStateType &molar_densities,
181  const StateType& T,
182  StateType& kfwd,
183  StateType& dkfwd_dT,
184  VectorStateType& dkfwd_dX) const
185  {
186  //variables, k0,kinf and derivatives
187  StateType k0 = Antioch::zero_clone(T);
188  StateType dk0_dT = Antioch::zero_clone(T);
189  StateType kinf = Antioch::zero_clone(T);
190  StateType dkinf_dT = Antioch::zero_clone(T);
191 
192  this->_forward_rate[0]->compute_rate_and_derivative(T,k0,dk0_dT);
193  this->_forward_rate[1]->compute_rate_and_derivative(T,kinf,dkinf_dT);
194 
195  StateType M = Antioch::zero_clone(T);
196  for(unsigned int i = 0; i < molar_densities.size(); i++)
197  {
198  M += molar_densities[i];
199  }
200 
201  //F
202  StateType f = Antioch::zero_clone(T);
203  StateType df_dT = Antioch::zero_clone(T);
204  VectorStateType df_dX = Antioch::zero_clone(molar_densities);
205  _F.F_and_derivatives(T,molar_densities,k0,dk0_dT,kinf,dkinf_dT,f,df_dT,df_dX);
206 
207 // k(T,[M]) = k0*[M]/(1 + [M]*k0/kinf) * F = k0 * ([M]^-1 + k0 * kinf^-1)^-1 * F
208  kfwd = k0 / (ant_pow(M,-1) + k0/kinf); //temp variable here for calculations dk_d{T,X}
209 
210 //dk_dT = F * dkfwd_dT + kfwd * dF_dT
211 // = F * kfwd * (dk0_dT/k0 - dk0_dT/(kinf/[M] + k0) + k0 * dkinf_dT/(kinf * (kinf/[M] + k0) ) )
212 // + dF_dT * kfwd
213  dkfwd_dT = f * kfwd * (dk0_dT/k0 - dk0_dT/(kinf/M + k0) + dkinf_dT * k0/(kinf * (kinf/M + k0)))
214  + df_dT * kfwd;
215 
216  dkfwd_dX.resize(this->n_species(), kfwd);
217 //dkfwd_dX = F * dkfwd_dX + kfwd * dF_dX
218 // = F * kfwd / ([M] + [M]^2 k0/kinf) + kfwd * dF_dX
219  for(unsigned int ic = 0; ic < this->n_species(); ic++)
220  {
221  dkfwd_dX[ic] = f * kfwd / (M + ant_pow(M,2) * k0/kinf) + df_dX[ic] * kfwd;
222  }
223 
224  kfwd *= f; //finalize
225 
226  return;
227  }
